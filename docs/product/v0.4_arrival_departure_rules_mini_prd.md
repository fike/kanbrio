## 💾 1. Database Schema & SQL DDL Migration

To support card-specific checklists and column-specific transit validation policies, the following PostgreSQL DDL migration must be implemented in `apps/api/migrations/20260529000001_arrival_departure_rules.sql`.

```sql
-- Migration: Add Card Checklists and Column Transition Rules

-- 1. Create card_checklists table
CREATE TABLE IF NOT EXISTS card_checklists (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    card_id UUID NOT NULL REFERENCES cards(id) ON DELETE CASCADE,
    title TEXT NOT NULL,
    is_completed BOOLEAN NOT NULL DEFAULT FALSE,
    position INT NOT NULL,
    completed_by UUID REFERENCES users(id) ON DELETE SET NULL,
    completed_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Index for retrieving card checklists ordered by position
CREATE INDEX IF NOT EXISTS idx_card_checklists_card_id_position ON card_checklists(card_id, position);

-- 2. Create transition_rules table
CREATE TABLE IF NOT EXISTS transition_rules (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    workspace_id UUID NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
    column_id UUID NOT NULL REFERENCES columns(id) ON DELETE CASCADE,
    rule_type VARCHAR(50) NOT NULL CHECK (rule_type IN ('arrival', 'departure')),
    criteria_type VARCHAR(50) NOT NULL CHECK (criteria_type IN ('assignee_required', 'checklist_completed', 'subtasks_completed')),
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Index to optimize rule lookups for a specific column and rule type
CREATE INDEX IF NOT EXISTS idx_transition_rules_column_rules ON transition_rules(column_id, rule_type);
```

### 2.2 Concurrency-Safe Transition Policy Engine (`Card::move_to_column`)

When a card is moved between columns, the core service transaction must perform optimistic checks on both departure policies of the source column and arrival policies of the target column.

```rust
impl Card {
    /// Validates policies and moves a card to a target column inside a database transaction.
    pub async fn move_to_column(
        pool: &sqlx::PgPool,
        workspace_id: Uuid,
        card_id: Uuid,
        target_column_id: Uuid,
        is_admin: bool,
        override_rules: bool,
    ) -> Result<Self, crate::AppError> {
        let mut tx = pool.begin().await?;

        // 1. Fetch current card and lock row
        let card = sqlx::query_as::<_, Card>(
            "SELECT * FROM cards WHERE id = $1 AND workspace_id = $2 FOR UPDATE"
        )
        .bind(card_id)
        .bind(workspace_id)
        .fetch_one(&mut *tx)
        .await?;

        let source_column_id = card.current_column_id;

        // Skip check if columns are identical or if admin override is passed
        if source_column_id != target_column_id && (!override_rules || !is_admin) {
            // 2. Validate Departure Rules (Source Column)
            let departure_rules: Vec<TransitionRule> = sqlx::query_as(
                "SELECT * FROM transition_rules WHERE column_id = $1 AND rule_type = 'departure'"
            )
            .bind(source_column_id)
            .fetch_all(&mut *tx)
            .await?;

            for rule in departure_rules {
                Self::evaluate_rule(&mut tx, &card, &rule).await?;
            }

            // 3. Validate Arrival Rules (Target Column)
            let arrival_rules: Vec<TransitionRule> = sqlx::query_as(
                "SELECT * FROM transition_rules WHERE column_id = $1 AND rule_type = 'arrival'"
            )
            .bind(target_column_id)
            .fetch_all(&mut *tx)
            .await?;

            for rule in arrival_rules {
                Self::evaluate_rule(&mut tx, &card, &rule).await?;
            }
        }

        // 4. Update column reference
        let updated_card = sqlx::query_as::<_, Card>(
            r#"
            UPDATE cards
            SET current_column_id = $1, updated_at = NOW()
            WHERE id = $2 AND workspace_id = $3
            RETURNING *
            "#
        )
        .bind(target_column_id)
        .bind(card_id)
        .bind(workspace_id)
        .fetch_one(&mut *tx)
        .await?;

        // 5. Record transition event
        let transition_type = if override_rules && is_admin {
            "move_override".to_string()
        } else {
            "move".to_string()
        };

        sqlx::query(
            r#"
            INSERT INTO card_transitions (card_id, transition_type, from_column_id, to_column_id)
            VALUES ($1, $2, $3, $4)
            "#
        )
        .bind(card_id)
        .bind(transition_type)
        .bind(source_column_id)
        .bind(target_column_id)
        .execute(&mut *tx)
        .await?;

        tx.commit().await?;
        Ok(updated_card)
    }

    async fn evaluate_rule(
        tx: &mut sqlx::Transaction<'_, sqlx::Postgres>,
        card: &Card,
        rule: &TransitionRule,
    ) -> Result<(), crate::AppError> {
        match rule.criteria_type.as_str() {
            "assignee_required" => {
                if card.assigned_user_id.is_none() {
                    return Err(crate::AppError::RuleViolation("Assignee is required".into()));
                }
            }
            "checklist_completed" => {
                let uncompleted_count: (i64,) = sqlx::query_as(
                    "SELECT COUNT(*) FROM card_checklists WHERE card_id = $1 AND is_completed = FALSE"
                )
                .bind(card.id)
                .fetch_one(&mut **tx)
                .await?;

                if uncompleted_count.0 > 0 {
                    return Err(crate::AppError::RuleViolation("All checklist items must be completed".into()));
                }
            }
            "subtasks_completed" => {
                // Assert that all child cards in the hierarchy are in a column marked `is_done = true`
                let active_children: (i64,) = sqlx::query_as(
                    r#"
                    SELECT COUNT(*)
                    FROM cards c
                    INNER JOIN columns col ON c.current_column_id = col.id
                    WHERE c.parent_id = $1 AND col.is_done = FALSE
                    "#
                )
                .bind(card.id)
                .fetch_one(&mut **tx)
                .await?;

                if active_children.0 > 0 {
                    return Err(crate::AppError::RuleViolation("All subtasks must be completed".into()));
                }
            }
            _ => {}
        }
        Ok(())
    }
}
```

```rust
#[sqlx::test]
async fn test_enforce_arrival_departure_rules(pool: sqlx::PgPool) -> anyhow::Result<()> {
    sqlx::migrate!("./migrations").run(&pool).await?;

    let workspace_id = Uuid::new_v4();
    let col_backlog = Uuid::new_v4();
    let col_review = Uuid::new_v4();
    let lane_id = Uuid::new_v4();

    // 1. Seed base components
    sqlx::query("INSERT INTO workspaces (id, name) VALUES ($1, 'Eng Workspace')").bind(workspace_id).execute(&pool).await?;
    sqlx::query("INSERT INTO columns (id, workspace_id, title, position) VALUES ($1, $2, 'Backlog', 1), ($3, $2, 'Review', 2)")
        .bind(col_backlog).bind(workspace_id).bind(col_review).execute(&pool).await?;
    sqlx::query("INSERT INTO swimlanes (id, workspace_id, title, position) VALUES ($1, $2, 'Standard', 1)").bind(lane_id).bind(workspace_id).execute(&pool).await?;

    // 2. Set Arrival Rule on 'Review' Column: Assignee Required
    sqlx::query("INSERT INTO transition_rules (workspace_id, column_id, rule_type, criteria_type) VALUES ($1, $2, 'arrival', 'assignee_required')")
        .bind(workspace_id).bind(col_review).execute(&pool).await?;

    // 3. Create card in Backlog (no assignee)
    let card_id = Uuid::new_v4();
    sqlx::query("INSERT INTO cards (id, workspace_id, title, current_column_id, current_swimlane_id) VALUES ($1, $2, 'Draft Card', $3, $4)")
        .bind(card_id).bind(workspace_id).bind(col_backlog).bind(lane_id).execute(&pool).await?;

    // 4. Try to move to 'Review' column without assignee (Should FAIL)
    let move_err = Card::move_to_column(&pool, workspace_id, card_id, col_review, false, false)
        .await
        .unwrap_err();

    assert!(matches!(move_err, AppError::RuleViolation(_)));

    Ok(())
}
```
